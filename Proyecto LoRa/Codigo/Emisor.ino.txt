#include <SPI.h>
// Librería SPI estándar para el bus del ESP32
#include <LoRa.h>
// Librería para el módulo Ra-02 (SX1278)
#include <Wire.h>
// Librería para el bus I2C del ESP32
#include <Adafruit_AHTX0.h>
// Librería para el sensor de temperatura y humedad

// --- CONFIGURACIÓN DE PINES Ra-02 PARA ESP32 ---
// Los pines SPI por defecto en el ESP32 (Wroom / DevKit v1) son:
// SCK  -> GPIO 18
// MISO -> GPIO 19
// MOSI -> GPIO 23
const int csPin = 5;       // NSS conectado al GPIO 5
const int resetPin = 14;   // RST conectado al GPIO 14
const int irqPin = 2;      // DIO0 conectado al GPIO 2 (¡Acá el LED azul va a parpadear al transmitir!)

Adafruit_AHTX0 aht; // Instancia del sensor AHT10/AHT20

void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("\n--- INICIALIZANDO NODO EMISOR ESP32 ---");

  // 1. Inicializar el sensor AHT20 en los pines I2C nativos del ESP32
  // En el ESP32, los pines I2C por defecto son: SDA = GPIO 21, SCL = GPIO 22
  Wire.begin(21, 22); 
  
  if (!aht.begin()) {
    Serial.println("ERROR: No se encuentra el sensor AHT20. Revisa cables I2C (GPIO 21 y 22).");
    while (1) delay(10); // Se detiene si no hay sensor
  }
  Serial.println("-> Sensor AHT20 detectado correctamente.");

  // 2. Configurar pines de control para el módulo LoRa
  LoRa.setPins(csPin, resetPin, irqPin);

  // 3. Iniciar el módulo LoRa en 433MHz
  if (!LoRa.begin(433E6)) {
    Serial.println("ERROR: No se encuentra el módulo LoRa. Revisa pines SPI.");
    while (1) delay(10); // Se detiene si no hay módulo de radio
  }

  // Palabra de sincronización (Debe ser idéntica en el receptor: 0xF3)
  LoRa.setSyncWord(0xF3);
  
  Serial.println("¡Nodo Emisor LoRa (ESP32) Listo y Transmitiendo!");
}

void loop() {
  // Lectura de variables del sensor
  sensors_event_t humidity, temp;
  aht.getEvent(&humidity, &temp);

  // Formatear el mensaje en una sola cadena de texto clara (Ej: "24.5C  62.3%")
  String mensaje = String(temp.temperature, 1) + "C  " + String(humidity.relative_humidity, 1) + "%";

  // Imprimir por el Monitor Serie de la compu para control visual
  Serial.print("Transmitiendo -> ");
  Serial.println(mensaje);

  // Enviar el paquete por el aire a través de LoRa
  LoRa.beginPacket();
  LoRa.print(mensaje);
  LoRa.endPacket();

  delay(4000); // Envía datos cada 4 segundos
}
